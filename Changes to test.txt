<template>
    <lightning-button label="New Case" onclick={handleClick}></lightning-button>
</template>

// newCaseWithConfirmation.js
import { LightningElement } from 'lwc';

export default class NewCaseWithConfirmation extends LightningElement {
    handleClick() {
        if (confirm('Do you want to create a new case?')) {
            // Redirect to the standard Case creation page
            window.location.href = '/lightning/o/Case/new'; // Redirect to the standard Case creation page
        }
    }
}
